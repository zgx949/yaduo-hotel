```python
import requests


headers = {
    'Host': 'api2.yaduo.com',
    'Accept': '*/*',
    'At-Platform-Type': '2',
    'At-Client-Id': '49725C6E-CC13-4EF5-91C9-A92A1A03B56D',
    'At-App-Version': '4.8.1',
    'Accept-Language': 'zh-Hans-CN;q=1',
    'Content-Type': 'application/json',
    'User-Agent': 'AtourLife/4.8.1 (iPhone; iOS 18.7.5; Scale/3.00)',
    'Connection': 'keep-alive',
    'At-Channel-Id': '20001',
}

params = {
    'platType': '2',
    'appVer': '4.8.1',
    'token': 'e1fa44c5e23c470c84c02d19680c59ecvwCkNfXG',
    'channelId': '20001',
}

response = requests.get(
    'https://api2.yaduo.com/atourlife/member/queryMarketingDataV2',
    params=params,
    headers=headers,
)

data = response.json()

print(data)
```
